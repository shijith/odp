116 static void clear_sched_queues(test_globals_t *globals)
117 {
118  odp_event_t ev;
119  odp_buffer_t buf;
120  test_event_t *event;
121  int i, j;
122  odp_queue_t fromq;
123 
124  /* Allocate the cool_down event. */
125  buf = odp_buffer_alloc(globals->pool);
126  if (buf == ODP_BUFFER_INVALID)
127  ODPH_ABORT("Buffer alloc failed.\n");
128 
129  event = odp_buffer_addr(buf);
130  event->type = COOL_DOWN;
131  ev = odp_buffer_to_event(buf);
132 
133  for (i = 0; i < NUM_PRIOS; i++) {
134  for (j = 0; j < globals->args.prio[i].queues; j++) {
135  /* Enqueue cool_down event on each queue. */
136  if (odp_queue_enq(globals->queue[i][j], ev))
137  ODPH_ABORT("Queue enqueue failed.\n");
138 
139  /* Invoke scheduler until cool_down event has been
140  * received. */
141  while (1) {
142  ev = odp_schedule(NULL, ODP_SCHED_WAIT);
143  buf = odp_buffer_from_event(ev);
144  event = odp_buffer_addr(buf);
145  if (event->type == COOL_DOWN)
146  break;
147  odp_event_free(ev);
148  }
149  }
150  }
151 
152  /* Free the cool_down event. */
153  odp_event_free(ev);
154 
155  /* Call odp_schedule() to trigger a release of any scheduler context. */
156  ev = odp_schedule(&fromq, ODP_SCHED_NO_WAIT);
157  if (ev != ODP_EVENT_INVALID)
158  ODPH_ABORT("Queue %" PRIu64 " not empty.\n",
159  odp_queue_to_u64(fromq));
160 }

176 static int enqueue_events(int prio, int num_queues, int num_events,
177  int num_samples, odp_bool_t div_events,
178  test_globals_t *globals)
179 {
180  odp_buffer_t buf[num_events + num_samples];
181  odp_event_t ev[num_events + num_samples];
182  odp_queue_t queue;
183  test_event_t *event;
184  int i, j, ret;
185  int enq_events;
186  int events_per_queue;
187  int tot_events;
188  int rdy_events = 0;
189 
190  tot_events = num_events + num_samples;
191 
192  if (!num_queues || !tot_events)
193  return 0;
194 
195  events_per_queue = tot_events;
196  if (div_events)
197  events_per_queue = (tot_events + num_queues - 1) / num_queues;
198 
199  for (i = 0; i < num_queues; i++) {
200  queue = globals->queue[prio][i];
201 
202  ret = odp_buffer_alloc_multi(globals->pool, buf,
203  events_per_queue);
204  if (ret != events_per_queue) {
205  ODPH_ERR("Buffer alloc failed. Try increasing EVENT_POOL_SIZE.\n");
206  ret = ret < 0 ? 0 : ret;
207  odp_buffer_free_multi(buf, ret);
208  return -1;
209  }
210  for (j = 0; j < events_per_queue; j++) {
211  if (!odp_buffer_is_valid(buf[j])) {
212  ODPH_ERR("Buffer alloc failed\n");
213  odp_buffer_free_multi(buf, events_per_queue);
214  return -1;
215  }
216 
217  event = odp_buffer_addr(buf[j]);
218  memset(event, 0, sizeof(test_event_t));
219 
220  /* Latency isn't measured from the first processing
221  * rounds. */
222  if (num_samples > 0) {
223  event->type = WARM_UP;
224  event->warm_up_rounds = 0;
225  num_samples--;
226  } else {
227  event->type = TRAFFIC;
228  }
229  event->src_idx[prio] = i;
230  event->prio = prio;
231  ev[j] = odp_buffer_to_event(buf[j]);
232  }
233 
234  enq_events = 0;
235  do {
236  ret = odp_queue_enq_multi(queue, &ev[enq_events],
237  events_per_queue -
238  enq_events);
239  if (ret < 0) {
240  ODPH_ERR("Queue enqueue failed.\n");
241  return -1;
242  }
243  enq_events += ret;
244  } while (enq_events < events_per_queue);
245 
246  rdy_events += events_per_queue;
247  if (div_events && rdy_events >= tot_events)
248  return 0;
249  }
250  return 0;
251 }

258 static void print_results(test_globals_t *globals)
259 {
260  test_stat_t *lat;
261  odp_schedule_sync_t stype;
262  test_stat_t total;
263  test_args_t *args;
264  uint64_t avg;

270  printf("\n%s queue scheduling latency\n",

272  ((stype == ODP_SCHED_SYNC_ORDERED) ? "ORDERED" : "PARALLEL"));
273 
274  printf(" Forwarding mode: %s\n",
275  (args->forward_mode == EVENT_FORWARD_RAND) ? "random" :
276  ((args->forward_mode == EVENT_FORWARD_INC) ? "incremental" :
277  "none"));
278 
279  printf(" LO_PRIO queues: %i\n", args->prio[LO_PRIO].queues);
280  if (args->prio[LO_PRIO].events_per_queue)
281  printf(" LO_PRIO event per queue: %i\n",
282  args->prio[LO_PRIO].events);
283  else
284  printf(" LO_PRIO events: %i\n", args->prio[LO_PRIO].events);
285 
286  printf(" LO_PRIO sample events: %i\n", args->prio[LO_PRIO].sample_events);
287 
288  printf(" HI_PRIO queues: %i\n", args->prio[HI_PRIO].queues);
289  if (args->prio[HI_PRIO].events_per_queue)
290  printf(" HI_PRIO event per queue: %i\n\n",

292  else
293  printf(" HI_PRIO events: %i\n", args->prio[HI_PRIO].events);
294 
295  printf(" HI_PRIO sample events: %i\n\n", args->prio[HI_PRIO].sample_events);
296 
297  for (i = 0; i < NUM_PRIOS; i++) {
298  memset(&total, 0, sizeof(test_stat_t));
299  total.min = UINT64_MAX;
300 
301  printf("%s priority\n"
302  "Thread Avg[ns] Min[ns] Max[ns] Samples Total Max idx\n"
303  "-----------------------------------------------------------------------\n",
304  i == HI_PRIO ? "HIGH" : "LOW");
305  for (j = 1; j <= args->cpu_count; j++) {
306  lat = &globals->core_stat[j].prio[i];
307 
308  if (lat->sample_events == 0) {
309  printf("%-8d N/A\n", j);
310  continue;
311  }
312 
313  if (lat->max > total.max)
314  total.max = lat->max;
315  if (lat->min < total.min)
316  total.min = lat->min;
317  total.tot += lat->tot;
318  total.sample_events += lat->sample_events;
319  total.events += lat->events;
320 
321  avg = lat->events ? lat->tot / lat->sample_events : 0;
322  printf("%-8d %-10" PRIu64 " %-10" PRIu64 " "
323  "%-10" PRIu64 " %-10" PRIu64 " %-10" PRIu64 " %-10" PRIu64 "\n",
324  j, avg, lat->min, lat->max, lat->sample_events,
325  lat->events, lat->max_idx);
326  }
327  printf("-----------------------------------------------------------------------\n");
328  if (total.sample_events == 0) {
329  printf("Total N/A\n\n");
330  continue;
331  }
332  avg = total.events ? total.tot / total.sample_events : 0;
333  printf("Total %-10" PRIu64 " %-10" PRIu64 " %-10" PRIu64 " "
334  "%-10" PRIu64 " %-10" PRIu64 "\n\n", avg, total.min,
335  total.max, total.sample_events, total.events);
336  }
337 }

339 static int join_groups(test_globals_t *globals, int thr)
340 {
341  odp_thrmask_t thrmask;
342  odp_schedule_group_t group;
343  int i, num;
344  int num_group = globals->args.num_group;
345 
346  if (num_group <= 0)
347  return 0;
348 
349  num = num_group;
350  if (globals->args.isolate)
351  num = 2 * num_group;
352 
353  odp_thrmask_zero(&thrmask);
354  odp_thrmask_set(&thrmask, thr);
355 
356  for (i = 0; i < num; i++) {
357  if (globals->args.isolate)
358  group = globals->group[i % 2][i / 2];
359  else
360  group = globals->group[0][i];
361 
362  if (odp_schedule_group_join(group, &thrmask)) {
363  ODPH_ERR("Group join failed %i (thr %i)\n", i, thr);
364  return -1;
365  }
366  }
367 
368  return 0;
369 }

388 static int test_schedule(int thr, test_globals_t *globals)
389 {
390  odp_time_t time;
391  odp_event_t ev;
392  odp_buffer_t buf;
393  odp_queue_t dst_queue;
394  uint64_t latency;
395  uint64_t i;
396  test_event_t *event;
397  test_stat_t *stats;
398  int dst_idx, change_queue;
399  int warm_up_rounds = globals->args.warm_up_rounds;
400  uint64_t test_rounds = globals->args.test_rounds * (uint64_t)1000000;
401 
402  memset(&globals->core_stat[thr], 0, sizeof(core_stat_t));
403  globals->core_stat[thr].prio[HI_PRIO].min = UINT64_MAX;
404  globals->core_stat[thr].prio[LO_PRIO].min = UINT64_MAX;
405 
406  change_queue = globals->args.forward_mode != EVENT_FORWARD_NONE ? 1 : 0;
407 
408  odp_barrier_wait(&globals->barrier);
409 
410  for (i = 0; i < test_rounds; i++) {
411  ev = odp_schedule(NULL, ODP_SCHED_WAIT);
412 
413  time = odp_time_global_strict();
414 
415  buf = odp_buffer_from_event(ev);
416  event = odp_buffer_addr(buf);
417 
418  stats = &globals->core_stat[thr].prio[event->prio];
419 
420  if (event->type == SAMPLE) {
421  latency = odp_time_to_ns(time) - odp_time_to_ns(event->time_stamp);
422 
423  if (latency > stats->max) {
424  stats->max = latency;
425  stats->max_idx = stats->sample_events;
426  }
427  if (latency < stats->min)
428  stats->min = latency;
429  stats->tot += latency;
430  stats->sample_events++;
431 
432  /* Move sample event to a different priority */
433  if (!globals->args.sample_per_prio &&
434  globals->args.prio[!event->prio].queues)
435  event->prio = !event->prio;
436  }
437 
438  if (odp_unlikely(event->type == WARM_UP)) {
439  event->warm_up_rounds++;
440  if (event->warm_up_rounds >= warm_up_rounds)
441  event->type = SAMPLE;
442  } else {
443  stats->events++;
444  }
445 
446  /* Move event to next queue if forwarding is enabled */
447  if (change_queue)
448  dst_idx = event->src_idx[event->prio] + 1;
449  else
450  dst_idx = event->src_idx[event->prio];
451  if (dst_idx >= globals->args.prio[event->prio].queues)
452  dst_idx = 0;
453  event->src_idx[event->prio] = dst_idx;
454  dst_queue = globals->queue[event->prio][dst_idx];
455 
456  if (event->type == SAMPLE)
457  event->time_stamp = odp_time_global_strict();
458 
459  if (odp_queue_enq(dst_queue, ev)) {
460  ODPH_ERR("[%i] Queue enqueue failed.\n", thr);
461  odp_event_free(ev);
462  return -1;
463  }
464  }
465 
466  /* Clear possible locally stored buffers */
467  odp_schedule_pause();
468 
469  while (1) {
470  odp_queue_t src_queue;
471 
472  ev = odp_schedule(&src_queue, ODP_SCHED_NO_WAIT);
473 
474  if (ev == ODP_EVENT_INVALID)
475  break;
476 
477  if (odp_queue_enq(src_queue, ev)) {
478  ODPH_ERR("[%i] Queue enqueue failed.\n", thr);
479  odp_event_free(ev);
480  return -1;
481  }
482  }
483 
484  odp_barrier_wait(&globals->barrier);
485 
486  if (thr == MAIN_THREAD) {
487  odp_schedule_resume();
488  clear_sched_queues(globals);
489  print_results(globals);
490  }
491 
492  return 0;
493 }

503 static int run_thread(void *arg ODP_UNUSED)
504 {
505  odp_shm_t shm;
506  test_globals_t *globals;
507  test_args_t *args;
508  int thr;
509 
510  thr = odp_thread_id();
511 
512  shm = odp_shm_lookup("test_globals");
513  globals = odp_shm_addr(shm);
514 
515  if (globals == NULL) {
516  ODPH_ERR("Shared mem lookup failed\n");
517  return -1;
518  }
519 
520  if (join_groups(globals, thr))
521  return -1;
522 
523  if (thr == MAIN_THREAD) {
524  args = &globals->args;
525 
526  if (enqueue_events(HI_PRIO, args->prio[HI_PRIO].queues,
527  args->prio[HI_PRIO].events, args->prio[HI_PRIO].sample_events,
528  !args->prio[HI_PRIO].events_per_queue,
529  globals))
530  return -1;
531 
532  if (enqueue_events(LO_PRIO, args->prio[LO_PRIO].queues,
533  args->prio[LO_PRIO].events, args->prio[LO_PRIO].sample_events,
534  !args->prio[LO_PRIO].events_per_queue,
535  globals))
536  return -1;
537  }
538 
539  if (test_schedule(thr, globals))
540  return -1;
541 
542  return 0;
543 }

739 static void randomize_queues(odp_queue_t queues[], uint32_t num, uint64_t *seed)
740 {
741  uint32_t i;
742 
743  for (i = 0; i < num; i++) {
744  uint32_t new_index;
745  odp_queue_t swap_queue;
746  odp_queue_t cur_queue = queues[i];
747 
748  odp_random_test_data((uint8_t *)&new_index, sizeof(new_index),
749  seed);
750  new_index = new_index % num;
751  swap_queue = queues[new_index];
752 
753  queues[new_index] = cur_queue;
754  queues[i] = swap_queue;
755  }

758 static int create_groups(test_globals_t *globals, odp_schedule_group_t group[], int num)
759 {

763

775  return 0;

788  ODPH_ERR("Group create failed %i\n", i);
789  break;
790  }

793  globals->group[i % 2][i / 2] = group[i];
794  } else {
795  globals->group[0][i] = group[i];
796  globals->group[1][i] = group[i];
797  }
798  }
799
